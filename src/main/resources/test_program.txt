((def "fibonacci")
  (if-var "<" norm ((param 0) (num 2)))
    (return (num 1))
  (else)
    (set-var "-" ((var "dif") (param 0) (num 1)))
    (call "fibonacci" ((var "dif")))
    (set-var "=" ((var "first") (ret)))
    (set-var "-" ((var "dif") (param 0) (num 2)))
    (call "fibonacci" ((var "dif")))
    (set-var "=" ((var "second") (ret)))
    (set-var "+" ((var "sum") (var "first") (var "second")))
    (return (var "sum"))
  (end-if))

((player-event "Join")
  (repeat "Multiple" "" norm ((var "i") (num 10)))
  (call "fibonacci" ((var "i")))
  (set-var "SetItemName" ((var "item") (item '{id:"minecraft:stone",Count:1}') (text "%var(^ret)")))
  (player-action "GiveItems" default ((var "item")))
  (end-repeat))
